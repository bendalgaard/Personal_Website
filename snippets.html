Bash Profile - editor colors 

		    						
black="\033[1;30m";
red="\033[1;31m";
green="\033[1;32m";
yellow="\033[1;33m";
blue="\033[1;34m";
magenta="\033[1;35m";
cyan="\033[1;36m";
white="\033[1;37m";
norm="\033[0;39m";

if [ "$PS1" ]; then
	if [[ $UID -eq 0 ]]; then
		PS1="\[$yellow\][\[\$(date +%H:%M:%S)\]] \[$red\]\u@\h:\w\\[$norm\] # "
	else
		PS1="\[$yellow\][\[\$(date +%H:%M:%S)\]] \[$cyan\]\u@\h:\w\\[$norm\] $ "
	fi
	export PS1=""$PS1"\[\e]30;\u@\h\a\]"
fi
